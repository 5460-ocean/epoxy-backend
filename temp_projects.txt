@router.get("/", response_model=list[ProjectOut])
def get_projects(
    skip: int = 0,
    limit: int = 10,
    name: str | None = None,
    surface: str | None = None,
    sort: str | None = None,
    db: Session = Depends(get_db),
    user=Depends(get_current_user)
):
    query = db.query(models.Project).filter(
        models.Project.owner_id == user.id,
        models.Project.is_deleted == False
    )

    # 🔍 Filtering
    if name:
        query = query.filter(models.Project.name.ilike(f"%{name}%"))

    if surface:
        query = query.filter(models.Project.surface.ilike(f"%{surface}%"))

    # 🔽 Sorting
    if sort:
        if sort.startswith("-"):
            field = sort[1:]
            query = query.order_by(getattr(models.Project, field).desc())
        else:
            query = query.order_by(getattr(models.Project, sort))

    # 📄 Pagination
    projects = query.offset(skip).limit(limit).all()

    return projects
