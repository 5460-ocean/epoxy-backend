function getStyleColors(style) { if (!style) return ["#00c6ff", "#003366"]; style = style.toLowerCase(); let colors = []; if (style.includes("ocean") || style.includes("water") || style.includes("sea")) { colors.push(["#00c6ff", "#003366"]); } if (style.includes("fire") || style.includes("lava")) { colors.push(["#ff512f", "#dd2476"]); } if (style.includes("sunset") || style.includes("sunrise")) { colors.push(["#ff9966", "#ff5e62"]); } if (style.includes("galaxy") || style.includes("space")) { colors.push(["#8e2de2", "#000000"]); } if (style.includes("forest") || style.includes("nature")) { colors.push(["#134e5e", "#71b280"]); } if (style.includes("marble") || style.includes("stone")) { colors.push(["#eeeeee", "#777777"]); } if (colors.length > 1) { return colors[Math.floor(Math.random() * colors.length)]; } if (colors.length === 1) { return colors[0]; } return ["#00c6ff", "#003366"]; }
