
// stronger luxury veins
float veins =
    smoothstep(0.68,0.73,veinField)
  - smoothstep(0.73,0.80,veinField);


// richer gold
col = mix(col,gold,veins*2.8);

// glowing metallic edge
col += veins * vec3(1.0,0.85,0.35) * 0.35;


// metallic flakes
float flakes =
    pow(fbm(p*25.0), 18.0);

col += flakes * vec3(0.8,0.9,1.0) * 0.15;
